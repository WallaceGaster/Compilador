main {
    int x, y, z;
    float a, b;
    
    x = 5;
    y = 3;
    z = (x + y) * 2 - (x / y);
    a = 2.5;
    b = (a ^ 2) + (3.5 * x) - 2;
    
    if z > 10 && b < 20 then
        cout << "Condición cumplida";
        z = z + 1;
        b = b / 2;
    end
    
    while x < 15
        x = x + (y * 2);
        y = y - 1;
    end
}
